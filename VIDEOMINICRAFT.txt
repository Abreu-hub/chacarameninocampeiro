<script>
  const senhaCorreta = 'chacarameninocampeiroprojeto';

  const senhaInput = document.getElementById('senhaInput');
  const senhaForm = document.getElementById('senhaForm');
  const videoForm = document.getElementById('videoForm');
  const btnEntrar = document.getElementById('btnEntrar');
  const btnAdicionar = document.getElementById('btnAdicionar');
  const btnSair = document.getElementById('btnSair');

  const urlInput = document.getElementById('urlInput');
  const serieInput = document.getElementById('serieInput');
  const nomeInput = document.getElementById('nomeInput');
  const videosContainer = document.getElementById('videosContainer');

  const videoModal = document.getElementById('videoModal');
  const videoIframe = document.getElementById('videoIframe');
  const closeModal = document.getElementById('closeModal');

  let videosData = {};

  function carregarVideos() {
    const data = localStorage.getItem('videosGerenciador');
    if (data) videosData = JSON.parse(data);
    else videosData = {};
  }

  function salvarVideos() {
    localStorage.setItem('videosGerenciador', JSON.stringify(videosData));
  }

  function extrairIdYoutube(url) {
    const match = url.match(/(?:v=|\/)([0-9A-Za-z_-]{11})/);
    return match ? match[1] : null;
  }

  function mostrarVideos() {
    videosContainer.innerHTML = '';
    const series = Object.keys(videosData);

    series.forEach(serie => {
      const section = document.createElement('section');
      section.className = 'series-section';

      const h2 = document.createElement('h2');
      h2.textContent = serie;
      section.appendChild(h2);

      const grid = document.createElement('div');
      grid.className = 'videos-grid';

      videosData[serie].forEach(video => {
        const card = document.createElement('div');
        card.className = 'video-card';

        const id = extrairIdYoutube(video.url);
        if (!id) return;

        const img = document.createElement('img');
        img.src = `https://img.youtube.com/vi/${id}/0.jpg`;
        img.alt = video.nome;
        img.addEventListener('click', () => {
          videoIframe.src = `https://www.youtube.com/embed/${id}?autoplay=1`;
          videoModal.style.display = 'flex';
        });

        const title = document.createElement('p');
        title.textContent = video.nome;

        card.appendChild(img);
        card.appendChild(title);
        grid.appendChild(card);
      });

      section.appendChild(grid);
      videosContainer.appendChild(section);
    });
  }

  btnEntrar.onclick = () => {
    if (senhaInput.value === senhaCorreta) {
      senhaForm.style.display = 'none';
      videoForm.style.display = 'block';
      carregarVideos();
      mostrarVideos();
    } else {
      alert('Senha incorreta!');
    }
  };

  btnAdicionar.onclick = () => {
    const url = urlInput.value.trim();
    const serie = serieInput.value.trim();
    const nome = nomeInput.value.trim();
    const id = extrairIdYoutube(url);

    if (!url || !serie || !nome || !id) {
      alert('Preencha todos os campos corretamente!');
      return;
    }

    if (!videosData[serie]) videosData[serie] = [];
    videosData[serie].push({ url, nome });

    salvarVideos();
    mostrarVideos();

    urlInput.value = '';
    serieInput.value = '';
    nomeInput.value = '';
  };

  btnSair.onclick = () => {
    videoForm.style.display = 'none';
    senhaForm.style.display = 'block';
  };

  closeModal.onclick = () => {
    videoModal.style.display = 'none';
    videoIframe.src = '';
  };

  window.onclick = (event) => {
    if (event.target == videoModal) {
      videoModal.style.display = 'none';
      videoIframe.src = '';
    }
  };

  carregarVideos();
  mostrarVideos();
</script>
