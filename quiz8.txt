  <script>
    const quizData = [
      {
        question: "O que o Creeper faz quando se aproxima do jogador?",
        options: ["Explode", "Foge", "Começa a cavar", "Atira flechas"],
        answer: 0
      },
      {
        question: "Qual é a fraqueza do Esqueleto?",
        options: ["Água", "Explosões", "Luz solar", "Fogo"],
        answer: 2
      },
      {
        question: "De qual mob o Creeper tem medo?",
        options: ["Esqueleto", "Aranha", "Gato/Ocelote", "Zumbi"],
        answer: 2
      },
      {
        question: "O Esqueleto ataca com qual arma?",
        options: ["Espada", "Arco e flecha", "Tridente", "Machado"],
        answer: 1
      },
      {
        question: "O que acontece se um Creeper for atingido por um raio?",
        options: ["Ele vira um zumbi", "Ele morre", "Ele vira um Creeper carregado", "Nada"],
        answer: 2
      },
      {
        question: "Qual mob pode dropar discos musicais se morto por um Esqueleto?",
        options: ["Zumbi", "Aranha", "Creeper", "Slime"],
        answer: 2
      },
      {
        question: "O Esqueleto é imune a qual destes efeitos?",
        options: ["Fogo", "Queda", "Veneno", "Lentidão"],
        answer: 2
      },
      {
        question: "Quantos corações de vida um Creeper tem?",
        options: ["10", "20", "15", "12"],
        answer: 1
      },
      {
        question: "Qual som o Creeper faz antes de explodir?",
        options: ["Grito", "Assobio", "Chiado", "Som de trovão"],
        answer: 2
      },
      {
        question: "O que acontece com o Esqueleto no sol sem capacete?",
        options: ["Fica mais rápido", "Brilha", "Queima", "Fica invisível"],
        answer: 2
      }
    ];

    let currentQuestion = 0;
    let score = 0;

    function loadQuestion() {
      const q = quizData[currentQuestion];
      document.getElementById("question").textContent = q.question;
      const optionsDiv = document.getElementById("options");
      optionsDiv.innerHTML = "";
      q.options.forEach((option, index) => {
        const btn = document.createElement("button");
        btn.textContent = option;
        btn.onclick = () => checkAnswer(index, btn);
        optionsDiv.appendChild(btn);
      });
    }

    function checkAnswer(selected, button) {
      const correctIndex = quizData[currentQuestion].answer;
      const buttons = document.querySelectorAll(".options button");
      buttons.forEach(btn => btn.classList.add("disabled"));
      if (selected === correctIndex) {
        button.classList.add("correct");
        score++;
      } else {
        buttons[correctIndex].classList.add("correct");
      }
      document.getElementById("next-btn").style.display = "inline-block";
    }

    function nextQuestion() {
      currentQuestion++;
      document.getElementById("next-btn").style.display = "none";
      if (currentQuestion < quizData.length) {
        loadQuestion();
      } else {
        showResult();
      }
    }

    function showResult() {
      document.getElementById("quiz").style.display = "none";
      let message = "";
      if (score <= 3) {
        message = "😢 Foi mal! Tente novamente.";
      } else if (score <= 5) {
        message = "😐 Precisa melhorar!";
      } else if (score <= 8) {
        message = "😊 Bom trabalho!";
      } else {
        message = "🏆 Muito bom! Você é fera no Minecraft!";
      }

      document.getElementById("result").innerHTML = `
        <p>Você acertou ${score} de ${quizData.length} perguntas.</p>
        <p>${message}</p>
        <button onclick="window.location.href='quiz.html'">Refazer Quiz</button>
      `;
    }

    // Inicia o quiz
    loadQuestion();
  </script>
